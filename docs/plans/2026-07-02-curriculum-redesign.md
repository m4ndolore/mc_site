**Architecture:** One root-level Vite MPA entry (like `start.html`). ### Task 2: Build-time injection script

**Files:**
- Create: `scripts/inject-curriculum.mjs`

**Step 1: Write the script**

It reads `data/curriculum.json` and `curriculum.html`, generates stage-section HTML, stage-rail HTML, and Course JSON-LD, and replaces content between marker comments in place. Idempotent — safe to re-run.

```js
// scripts/inject-curriculum.mjs — inject curriculum content as static HTML
// Follows the optimize-static.mjs pattern: mutates source HTML in place at
// build time so crawlers and no-JS visitors get the full syllabus.
import { readFileSync, writeFileSync } from 'fs'
import { resolve } from 'path'

const ROOT = resolve(import.meta.dirname, '..')
const PAGE = resolve(ROOT, 'curriculum.html')
const DATA = resolve(ROOT, 'data/curriculum.json')

const esc = (s) => String(s ?? '')
  .replace(/&/g, '&amp;').replace(/</g, '&lt;').replace(/>/g, '&gt;')
  .replace(/"/g, '&quot;').replace(/'/g, '&#39;')

const TYPE_LABEL = { external: 'External', gated: 'Gated', learn: 'Learn', signal: 'Signal' }

const curriculum = JSON.parse(readFileSync(DATA, 'utf8'))
const stages = curriculum.stages

function resourceRow(r, stageId) {
  const external = r.type === 'external'
  const meta = [r.sourceLabel, `${r.timeToRead} min`, TYPE_LABEL[r.type] || r.type]
    .filter(Boolean).map(esc).join(' · ')
  return `
        <a class="cur-resource" data-resource-id="${esc(r.id)}" data-stage-id="${esc(stageId)}"
           href="${esc(r.url)}"${external ? ' target="_blank" rel="noopener"' : ''}>
          <span class="cur-resource__check" aria-hidden="true"></span>
          <span class="cur-resource__body">
            <span class="cur-resource__title">${esc(r.title)}</span>
            <span class="cur-resource__desc">${esc(r.description)}</span>
            <span class="cur-resource__meta">${meta}</span>
          </span>
          <span class="cur-resource__open">Open ${external ? '↗' : '→'}</span>
        </a>`
}

function stageSection(s, i) {
  const num = String(i + 1).padStart(2, '0')
  return `
      <section class="cur-stage" data-stage-id="${esc(s.id)}" data-stage-index="${i}" id="stage-${esc(s.id)}">
        <div class="cur-stage__head">
          <span class="cur-stage__num">${num}</span>
          <h2 class="cur-stage__title">${esc(s.title)}</h2>
          <p class="cur-stage__subtitle">${esc(s.subtitle)}</p>
        </div>
        <p class="cur-stage__desc">${esc(s.description)}</p>
        <div class="cur-stage__resources">${s.resources.map(r => resourceRow(r, s.id)).join('')}
        </div>
        <div class="cur-stage__actions">
          <p class="cur-stage__count" data-count-for="${esc(s.id)}"></p>
          <button class="fp-btn fp-btn--primary cur-advance" data-advance-from="${esc(s.id)}" hidden></button>
        </div>
      </section>`
}

function railItem(s, i) {
  return `
              <li><a href="#stage-${esc(s.id)}" class="cur-rail__link" data-rail-stage="${esc(s.id)}">
                <span class="cur-rail__marker" aria-hidden="true"></span>
                <span class="cur-rail__label">${i + 1}. ${esc(s.title)}</span>
              </a></li>`
}

const jsonld = JSON.stringify({
  '@context': 'https://schema.org',
  '@type': 'Course',
  name: 'Defense Builders Curriculum',
  description: 'Progressive five-stage curriculum for defense tech founders: Preflight, Spot, Ready for Launch, Tension, Launch.',
  url: 'https://mergecombinator.com/curriculum',
  provider: { '@type': 'Organization', name: 'Merge Combinator', url: 'https://mergecombinator.com' },
  hasPart: stages.map((s, i) => ({
    '@type': 'ItemList',
    name: `Stage ${i + 1}: ${s.title}`,
    description: s.description,
    itemListElement: s.resources.map((r, j) => ({
      '@type': 'ListItem', position: j + 1, name: r.title, url: r.url,
    })),
  })),
}, null, 2)

function inject(html, tag, content) {
  const re = new RegExp(`(<!-- CURRICULUM:${tag}:START -->)[\\s\\S]*?(<!-- CURRICULUM:${tag}:END -->)`)
  if (!re.test(html)) throw new Error(`Marker CURRICULUM:${tag} not found in curriculum.html`)
  return html.replace(re, `$1${content}\n    $2`)
}

let html = readFileSync(PAGE, 'utf8')
html = inject(html, 'CONTENT', stages.map(stageSection).join('\n'))
html = inject(html, 'RAIL', stages.map(railItem).join(''))
html = inject(html, 'JSONLD', `\n<script type="application/ld+json">\n${jsonld}\n</script>`)
writeFileSync(PAGE, html)

const total = stages.reduce((n, s) => n + s.resources.length, 0)
console.log(`[curriculum] injected ${stages.length} stages, ${total} resources into curriculum.html`)
```

```js
// js/curriculum.js — progressive enhancement over the build-time-injected
// curriculum DOM. State in localStorage; metrics via Plausible.
const STORAGE_KEY = 'mc.curriculum.v1'
const TRIAGE_KEY = 'mc-founder-path-state-v1'
const ADVANCE_GATE = 2

const TRIAGE_TO_STAGE = {
  'visionary-no-problem': 'preflight',
  'curious': 'preflight',
  'operator-with-problem': 'spot',
  'builder-no-problem': 'spot',
  'team-with-prototype': 'ready',
  'scaling': 'tension',
}

const TRIAGE_LABELS = {
  'visionary-no-problem': 'a visionary without a problem yet',
  'curious': 'exploring defense tech',
  'operator-with-problem': 'an operator with a problem',
  'builder-no-problem': 'a technical builder seeking a problem',
  'team-with-prototype': 'a team with a working prototype',
  'scaling': 'fundraising, scaling, or transitioning',
}

function track(event, props) {
  const p = window.plausible
  if (typeof p === 'function') p(event, props ? { props } : undefined)
}

const stagesEl = document.getElementById('curStages')
const stageEls = [...document.querySelectorAll('.cur-stage')]
const stageIds = stageEls.map(el => el.dataset.stageId)
if (!stageEls.length) throw new Error('curriculum content missing — was inject-curriculum.mjs run?')

function loadState() {
  try {
    const s = JSON.parse(localStorage.getItem(STORAGE_KEY))
    if (s && stageIds.includes(s.currentStage)) {
      return { currentStage: s.currentStage, engaged: s.engaged || [], startedFrom: s.startedFrom || null, bannerDismissed: !!s.bannerDismissed }
    }
  } catch { /* ignore */ }
  return null
}

function triageStage() {
  try {
    const t = JSON.parse(localStorage.getItem(TRIAGE_KEY))
    if (t && t.stage && TRIAGE_TO_STAGE[t.stage]) return { stage: TRIAGE_TO_STAGE[t.stage], answer: t.stage }
  } catch { /* ignore */ }
  return null
}

let state = loadState()
let returning = !!state
if (!state) {
  const mapped = triageStage()
  state = {
    currentStage: mapped ? mapped.stage : stageIds[0],
    engaged: [],
    startedFrom: mapped ? mapped.answer : null,
    bannerDismissed: false,
  }
  track('Curriculum Start', { stage: state.currentStage, fromTriage: String(!!mapped) })
}
let viewingStage = state.currentStage

function save() {
  try { localStorage.setItem(STORAGE_KEY, JSON.stringify(state)) } catch { /* ignore */ }
}

const idx = (id) => stageIds.indexOf(id)
const engagedInStage = (id) =>
  [...document.querySelectorAll(`.cur-resource[data-stage-id="${id}"]`)]
    .filter(el => state.engaged.includes(el.dataset.resourceId)).length

function render() {
  const currentIdx = idx(state.currentStage)
  const viewIdx = idx(viewingStage)

  stageEls.forEach((el, i) => el.classList.toggle('is-open', i === viewIdx))

  // Progress segments: complete for passed stages, active for current
  document.querySelectorAll('#curProgress .fp-progress__seg').forEach((seg, i) => {
    seg.classList.toggle('is-complete', i < currentIdx)
    seg.classList.toggle('is-active', i === currentIdx)
  })
  const stageEl = stageEls[viewIdx]
  document.getElementById('curProgressLabel').textContent =
    `Stage ${currentIdx + 1} of ${stageIds.length} · ${stageEl.querySelector('.cur-stage__title').textContent}${viewIdx !== currentIdx ? ' (previewing ' + (viewIdx + 1) + ')' : ''}`

  // Rail states
  document.querySelectorAll('.cur-rail__link').forEach(link => {
    const i = idx(link.dataset.railStage)
    link.classList.toggle('is-complete', i < currentIdx)
    link.classList.toggle('is-current', i === currentIdx)
    link.classList.toggle('is-upcoming', i > currentIdx)
  })

  // Engagement marks
  document.querySelectorAll('.cur-resource').forEach(el => {
    el.classList.toggle('is-engaged', state.engaged.includes(el.dataset.resourceId))
  })

  // Per-stage count + advance button (only on the CURRENT stage)
  stageEls.forEach((el, i) => {
    const id = el.dataset.stageId
    const n = engagedInStage(id)
    const total = el.querySelectorAll('.cur-resource').length
    const countEl = el.querySelector('.cur-stage__count')
    const btn = el.querySelector('.cur-advance')
    countEl.textContent = `${n} of ${total} explored`
    const isCurrent = i === currentIdx
    const isLast = i === stageIds.length - 1
    btn.hidden = !isCurrent || isLast
    if (isCurrent && !isLast) {
      const nextTitle = stageEls[i + 1].querySelector('.cur-stage__title').textContent
      const unlocked = n >= ADVANCE_GATE
      btn.disabled = !unlocked
      btn.textContent = unlocked
        ? `Advance to ${nextTitle} →`
        : `Explore ${ADVANCE_GATE - n} more to advance`
    }
    if (isCurrent && isLast) {
      countEl.textContent += ' · Final stage'
    }
  })

  // Peek strip
  const peeking = viewIdx !== currentIdx
  const strip = document.getElementById('curPeekStrip')
  strip.hidden = !peeking
  if (peeking) {
    document.getElementById('curPeekText').textContent =
      `Your current stage is ${stageEls[currentIdx].querySelector('.cur-stage__title').textContent}.`
  }
}

function showBanner() {
  if (state.bannerDismissed || returning || !state.startedFrom) return
  if (state.currentStage === stageIds[0]) return
  const banner = document.getElementById('curTriageBanner')
  document.getElementById('curTriageBannerText').textContent =
    `You're ${TRIAGE_LABELS[state.startedFrom] || 'past the basics'} — we've started you at ${stageEls[idx(state.currentStage)].querySelector('.cur-stage__title').textContent}.`
  banner.hidden = false
}

// ── Events ──
stagesEl.addEventListener('click', (e) => {
  const check = e.target.closest('.cur-resource__check')
  if (check) {
    e.preventDefault()
    const card = check.closest('.cur-resource')
    const id = card.dataset.resourceId
    const on = !state.engaged.includes(id)
    state.engaged = on ? [...state.engaged, id] : state.engaged.filter(x => x !== id)
    if (on) track('Curriculum Resource Engaged', { resource: id, via: 'toggle' })
    save(); render()
    return
  }
  const card = e.target.closest('.cur-resource')
  if (card) {
    const id = card.dataset.resourceId
    if (!state.engaged.includes(id)) {
      state.engaged = [...state.engaged, id]
      track('Curriculum Resource Engaged', { resource: id, via: 'open' })
      save(); render()
    }
    track('Curriculum Resource Open', { resource: id })
    return // let the link navigate (new tab for external)
  }
  const adv = e.target.closest('.cur-advance')
  if (adv && !adv.disabled) {
    const from = state.currentStage
    const next = stageIds[idx(from) + 1]
    state.currentStage = next
    viewingStage = next
    track('Curriculum Advance', { from, to: next })
    save(); render()
    document.getElementById('cur-main').scrollIntoView({ behavior: 'smooth' })
  }
})

document.getElementById('curRail').addEventListener('click', (e) => {
  const link = e.target.closest('.cur-rail__link')
  if (!link) return
  e.preventDefault()
  viewingStage = link.dataset.railStage
  if (viewingStage !== state.currentStage) track('Curriculum Peek', { stage: viewingStage })
  render()
})

document.getElementById('curPeekReturn').addEventListener('click', () => {
  viewingStage = state.currentStage
  render()
})

document.getElementById('curBannerDismiss').addEventListener('click', () => {
  state.bannerDismissed = true
  save()
  document.getElementById('curTriageBanner').hidden = true
})

document.getElementById('curStartOver').addEventListener('click', () => {
  state.currentStage = stageIds[0]
  state.bannerDismissed = true
  viewingStage = stageIds[0]
  save(); render()
  document.getElementById('curTriageBanner').hidden = true
})

// ── Boot ──
stagesEl.classList.add('js-enhanced')
showBanner()
render()
```
